local function hasFileSystemSupport()
    return type(writefile) == "function" and type(readfile) == "function" and type(isfile) == "function"
end

local function saveVerifiedKey(key)
    if hasFileSystemSupport() then pcall(function() writefile("PanduHub_key.txt", key) end) end
end

local function loadVerifiedKey()
    if not hasFileSystemSupport() then return nil end
    local ok, content = pcall(function() return readfile("PanduHub_key.txt") end)
    return ok and content or nil
end

local function clearSavedKey()
    if hasFileSystemSupport() then pcall(function() delfile("PanduHub_key.txt") end) end
end

local function AuthenticatePremium()
    local savedKey = loadVerifiedKey()
    local keyToCheck = getgenv().SCRIPT_KEY

    if not keyToCheck or keyToCheck == "" or keyToCheck == "MASUKKAN_KEY_DISINI" then
        local link = Junkie.get_key_link()
        if link and setclipboard then setclipboard(link) end
        game:GetService("Players").LocalPlayer:Kick("\n[PANDU HUB PREMIUM]\nKEY REQUIRED!")
        return false
    end

    local result = Junkie.check_key(keyToCheck)

    if result and result.valid then
        if result.message == "KEYLESS" then
            getgenv().SCRIPT_KEY = "KEYLESS"
            return true
        elseif result.message == "KEY_VALID" then
            if not savedKey then saveVerifiedKey(keyToCheck) end
            getgenv().SCRIPT_KEY = keyToCheck
            return true
        end
    else
        if savedKey then 
            clearSavedKey() 
        end
        
        local link = Junkie.get_key_link()
        if link and setclipboard then setclipboard(link) end
        game:GetService("Players").LocalPlayer:Kick("\n[PANDU HUB PREMIUM]\nScript used on another HWID / Expired Key")
        return false
    end
end

local function censorUsername(name)
    if #name <= 2 then
        return string.rep("*", #name)
    end
    return name:sub(1,2) .. string.rep("*", #name - 2)
end

local function GetCharacterRoot()
    local char = Players.LocalPlayer.Character
    if char then
        return char:FindFirstChild("HumanoidRootPart")
    end
    return nil
end

local function TeleportToEscape()
    local root = GetCharacterRoot()
    local map = workspace:FindFirstChild("Map")
    if not root or not map then return false end
    
    local exitPos = nil
    
    pcall(function()
        for _, obj in ipairs(map:GetDescendants()) do
            local nameLower = string.lower(obj.Name)
            if nameLower == "fininshline" or nameLower == "finishline" or string.find(nameLower, "finish") then
                if obj:IsA("BasePart") then
                    exitPos = obj.Position
                    break
                elseif obj:IsA("Model") then
                    local part = obj:FindFirstChildWhichIsA("BasePart")
                    if part then
                        exitPos = part.Position
                        break
                    end
                end
            end
        end

        if not exitPos then
            if map:FindFirstChild("RooftopHitbox") or map:FindFirstChild("Rooftop") then 
                exitPos = Vector3.new(3098.16, 454.04, -4918.74)
            elseif map:FindFirstChild("HooksMeat") then 
                exitPos = Vector3.new(1546.12, 152.21, -796.72)
            elseif map:FindFirstChild("churchbell") then 
                exitPos = Vector3.new(760.98, -20.14, -78.48)
            else
                for _, obj in ipairs(map:GetDescendants()) do 
                    if obj:IsA("MeshPart") and obj.Material == Enum.Material.Limestone then 
                        exitPos = Vector3.new(-947.90, 152.12, -7579.52)
                        break 
                    elseif obj:IsA("MeshPart") and obj.Material == Enum.Material.Leather then 
                        exitPos = Vector3.new(1546.12, 152.21, -796.72)
                        break 
                    end 
                end
            end
        end
    end)
    
    if exitPos then 
        root.CFrame = CFrame.new(exitPos + Vector3.new(0, 3, 0)) 
        return true
    end
    
    return false
end

local function RandomServerHop()
    local success, result = pcall(function()
        return game:HttpGet(
            "https://games.roblox.com/v1/games/" ..
            tostring(game.PlaceId) ..
            "/servers/Public?sortOrder=Desc&limit=100&excludeFullGames=true"
        )
    end)

    if success and result then
        local data = HttpService:JSONDecode(result)
        local validServers = {}

        if data and data.data then
            for _, server in ipairs(data.data) do
                if type(server) == "table"
                    and server.id
                    and server.id ~= game.JobId then

                    table.insert(validServers, server.id)
                end
            end
        end

        if #validServers > 0 then
            local randomServer = validServers[math.random(#validServers)]
            game:GetService("TeleportService"):TeleportToPlaceInstance(
                game.PlaceId,
                randomServer,
                LocalPlayer
            )
        else
            if getgenv().Library then
                getgenv().Library:Notify("Gagal: Tidak ada server yang tersedia.", 3)
            end
        end
    end
end
local function HandleAFKLogic(currentTeam)
    if not Config.AUTO_AFKEscape then return end
    if AFK_Thread then task.cancel(AFK_Thread) end

    if currentTeam == "Survivors" then
        AFK_Thread = task.spawn(function()
            if Library then Library:Notify("AFK Mode ON!", 3) end
            task.wait(18)
            
            if Config.AUTO_AFKEscape and LocalPlayer.Team and LocalPlayer.Team.Name == "Survivors" then
                if TeleportToEscape() then
                    if Library then Library:Notify("Berhasil teleport ke exit! Menunggu server...", 3) end
                else
                    if Library then Library:Notify("Gagal menemukan exit!", 3) end
                end
            end
            
            task.wait(5)
            if Config.AUTO_AFKEscape and LocalPlayer.Team and LocalPlayer.Team.Name == "Survivors" then
                if Library then Library:Notify("Match (Stuck), mencari server baru...", 3) end
                RandomServerHop()
            end
        end)

    elseif currentTeam == "Spectator" then
        AFK_Thread = task.spawn(function()
            if PreviousTeam == "Survivors" then
                if Library then Library:Notify("Match selesai, mengirim data & mencari server baru...", 3) end
                
                SendWebhook()
                task.wait(5)
                RandomServerHop()
            else
                local isMatchOngoing = false
                for _, p in ipairs(Players:GetPlayers()) do
                    if p.Team and (p.Team.Name == "Survivors" or p.Team.Name == "Killer") then
                        isMatchOngoing = true
                        break
                    end
                end

                if isMatchOngoing then
                    if Library then Library:Notify("Match sedang berlangsung, mencari server baru...", 3) end
                    task.wait(2)
                    RandomServerHop()
                else
                    if Library then Library:Notify("Menunggu match mulai (Max 2 menit)..", 3) end
                    task.wait(120) 
                    if Library then Library:Notify("Waktu tunggu habis, mencari server baru...", 3) end
                    task.wait(2)
                    RandomServerHop()
                end
            end
        end)

    elseif currentTeam == "Killer" then
        AFK_Thread = task.spawn(function()
            if Library then Library:Notify("Kamu Killer! Mencari server baru...", 3) end
            task.wait(3)
            RandomServerHop()
        end)
    end
    
    PreviousTeam = currentTeam
end

            end
            task.wait(2)
            RandomServerHop()
        end
    end)

    pcall(function()
        local promptOverlay = CoreGui:WaitForChild("RobloxPromptGui", 5):WaitForChild("promptOverlay", 5)
        if promptOverlay then
            promptOverlay.ChildAdded:Connect(function(child)
                if Config.AUTO_AFKEscape and child.Name == "ErrorPrompt" then
                    if getgenv().Library then 
                        getgenv().Library:Notify("Server Penuh/Disconnect! Mencoba server hop...", 3) 
                    end
                    
                    pcall(function() child.Visible = false end)
                    
                    task.wait(2.5)
                    RandomServerHop()
                end
            end)
        end
    end)
end
